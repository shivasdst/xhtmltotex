\bookTitle{ನಾರಾಯಣ ಶಬ್ದಾರ್ಥಃ}

\centering{\Large(ಬಿದರಹಳ್ಳಿ ಶ್ರೀನಿವಾಸತೀರ್ಥ ವಿರಚಿತಃ)}

\vfill

\begin{center}
ನ ಮಾಧವಸಮೋ ದೇವೋ ನ ಚ ಮಧ್ವಸಮೋ ಗುರುಃ ।

ನ ತದ್ವಾಕ್ಯಸಮಂ ಶಾಸ್ತ್ರಂ ನ ಚ ತಸ್ಯ ಸಮಃ ಪುಮಾನ್ ।।
\end{center}

\vfill

\centering{ಕನ್ನಡ ಅನುವಾದ :}

\titleauthor{ಏರೀ ಸುಬ್ಬಣ್ಣಾಚಾರ್ಯ}

\centering{(ನಿವೃತ್ತ ಗೆಜೆಟೆಡ್ ಪೋಸ್ಟ್ ಮಾಸ್ಟರ್, ಚೆನ್ನೈ)}

\vfill

\centering{ಪ್ರಕಾಶಕರು :}

\medskip

\centering{\Large\bf ಶ್ರೀ ಸರ್ವಜ್ಞಾಚಾರ್ಯ ಸೇವಾ ಸಂಘ (ರಿ.)}

\smallskip

\centerline{ದಾವಣಗೆರೆ - ೫೭೭ ೦೦೨}
}
